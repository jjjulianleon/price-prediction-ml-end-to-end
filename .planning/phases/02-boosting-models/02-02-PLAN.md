The new comparison loop must follow this logic (adapt variable names to match what already exists in the file):

```python
# Fit preprocessor once on train sample
preprocessor = get_feature_pipeline()
X_train, y_train = split_features_target(train_df)
preprocessor.fit(X_train)

all_models_metrics: dict[str, dict[str, float]] = {}
candidates: dict[str, tuple] = {}  # name -> (model, preprocessor)

for entry in MODEL_ZOO:
    # --- TRAIN ---
    if entry.needs_incremental:
        # SGD out-of-core path: partial_fit over train_df in chunks
        # Since train_local works with a local DataFrame (not Snowflake),
        # simulate batching by splitting train_df into chunks of 50_000 rows.
        import math
        model = entry.estimator  # already configured in model_zoo
        chunk_size = 50_000
        n_chunks = max(1, math.ceil(len(train_df) / chunk_size))
        for i in range(n_chunks):
            chunk = train_df.iloc[i * chunk_size : (i + 1) * chunk_size]
            X_chunk, y_chunk = split_features_target(chunk)
            X_transformed = preprocessor.transform(X_chunk)
            model.partial_fit(X_transformed, y_chunk)
    else:
        # sample_only path: fit on full train_df (local data is already a sample)
        model = entry.estimator
        X_train_t = preprocessor.transform(X_train)
        model.fit(X_train_t, y_train)

    # --- EVALUATE ---
    # Use the existing evaluate_model_on_df function (or equivalent) that takes
    # model, preprocessor, and a DataFrame and returns RMSE.
    # If the function signature differs, adapt accordingly.
    val_rmse = evaluate_model_on_df(model, preprocessor, val_df)
    test_rmse = evaluate_model_on_df(model, preprocessor, test_df)

    metrics = {"val_rmse": val_rmse, "test_rmse": test_rmse}
    all_models_metrics[entry.name] = metrics
    candidates[entry.name] = (model, preprocessor, metrics)

# --- SELECT BEST ---
best_name = min(candidates, key=lambda n: candidates[n][2]["val_rmse"])
best_model, best_preprocessor, best_metrics = candidates[best_name]
```
